package saijava;
import java.util.Scanner;
public class practice {
	public static void main(String[] arg) {
		Scanner input=new Scanner(System.in);
		int n=input.nextInt();
		int i,j;
		for(i=0;i<n;i++)
		{
			for(j=0;j<=i;j++)
			{
				System.out.print("* ");
			}
			System.out.println();
		}
	}

}
